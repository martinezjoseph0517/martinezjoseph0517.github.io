# FUNCTION: Map cities to counties and aggregate
aggregate_cities_to_counties <- function(df, tax_col) {
  # Rename first column to City
  df <- df %>% rename(City = 1)
  df <- df %>% mutate(City = trimws(City))
  
  # Remove any rows where City is NA or total/summary rows
  df <- df %>% filter(!is.na(City), !grepl("total|statewide", City, ignore.case = TRUE))
  
  # Convert tax column to numeric (in case it's character)
  df[[tax_col]] <- as.numeric(df[[tax_col]])
  
  # Create city-to-county mapping for California
  # This uses tigris to get place (city) geometries and find which county they're in
  ca_places <- places(state = "CA", cb = TRUE, class = "sf")
  ca_counties_full <- counties(state = "CA", cb = TRUE, class = "sf")
  
  # Get centroids of places and spatial join to counties
  place_centroids <- st_centroid(ca_places)
  place_county_map <- st_join(place_centroids, ca_counties_full) %>%
    st_drop_geometry() %>%
    select(place_name = NAME.x, county_name = NAME.y) %>%
    distinct()
  
  # Try to match cities in your data to the place names
  df <- df %>%
    mutate(
      # Clean city names for better matching
      City_clean = gsub(" city| town| City| Town", "", City, ignore.case = TRUE),
      City_clean = trimws(City_clean)
    )
  
  # Join with county mapping
  df_mapped <- df %>%
    left_join(place_county_map, by = c("City_clean" = "place_name"))
  
  # For unmatched cities, try fuzzy matching or manual fixes
  unmatched <- df_mapped %>% filter(is.na(county_name))
  if (nrow(unmatched) > 0) {
    cat("Unmatched cities:", paste(unmatched$City, collapse = ", "), "\n")
  }
  
  # Aggregate by county
  df_aggregated <- df_mapped %>%
    filter(!is.na(county_name)) %>%
    group_by(County = county_name) %>%
    summarise(!!tax_col := sum(.data[[tax_col]], na.rm = TRUE), .groups = "drop")
  
  return(df_aggregated)
}

# Update FUNCTION: Create heatmap for one sheet
make_heatmap <- function(df, sheet_name, tax_col) {
  # Select relevant columns
  df <- df %>% select(County, all_of(tax_col))
  
  # Join with map data
  map_data <- ca_counties %>%
    left_join(df, by = "County")
  
  # Plot
  p <- ggplot(map_data) +
    geom_sf(aes(fill = .data[[tax_col]]), color = "white", size = 0.2) +
    scale_fill_viridis_c(option = "plasma", na.value = "grey90") +
    labs(
      title = paste("California TOT Heat Map —", sheet_name),
      fill = "TOT ($)"
    ) +
    theme_minimal()
  
  print(p)
}

  cat("\n=== Processing sheet", i, ":", sheet, "===\n")

  if (i %in% 1:4) {

    df <- aggregate_cities_to_counties(df, tax_col)
  } else {
    tax_col <- colnames(df)[grep("Transient Occupancy", colnames(df), ignore.case = TRUE)][1]
    
    df <- df %>%
      select(1, all_of(tax_col))
    
    colnames(df)[1] <- "County"
    
    df <- df %>%
      mutate(County = trimws(County)) %>%
      filter(!grepl("grand total|total|statewide", County, ignore.case = TRUE))
  }

  df[[tax_col]] <- as.numeric(df[[tax_col]])

  # Get top 10
  top_10 <- df %>%
    filter(!is.na(.data[[tax_col]]), .data[[tax_col]] > 0) %>%
    arrange(desc(.data[[tax_col]])) %>%
    slice_head(n = 10) %>%
    mutate(Sheet = sheet, Revenue = .data[[tax_col]]) %>%
    select(Sheet, County, Revenue)

  all_top_counties <- bind_rows(all_top_counties, top_10)
}
